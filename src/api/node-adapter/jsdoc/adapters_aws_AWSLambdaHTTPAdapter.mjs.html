adapters/aws/AWSLambdaHTTPAdapter.mjs

import mime from 'mime'
import accepts from 'accepts'
import statuses from 'statuses'
import { Adapter } from '../Adapter.mjs'
import { OutputMapper } from '../../mappers/OutputMapper.mjs'
import { AWS_LAMBDA_HTTP_PLATFORM } from '../../constants.mjs'
import { HttpEventMapper } from '../../mappers/HttpEventMapper.mjs'
import { isFunction, isBrowser, RuntimeError } from '@stone-js/common'
import { inputMiddleware, outputMiddleware } from '../../middleware/aws/http/index.mjs'

/**
 * Class representing a AWSLambdaHTTPAdapter.
 *
 * @author Mr. Stone <evensstone@gmail.com>
 *
 * @extends Adapter
 */
export class AWSLambdaHTTPAdapter extends Adapter {
  /**
   * Create an Apapter.
   *
   * @param {adapterOptions} [options={}] - Apapter configuration options.
   */
  constructor (options = {}) {
    super(options)

    this
      .registerServices(inputMiddleware.concat(outputMiddleware))
      .setInputMapper(HttpEventMapper.create(this.container, inputMiddleware))
      .setOutputMapper(OutputMapper.create(this.container, outputMiddleware))
      .addContext({
        platform: AWS_LAMBDA_HTTP_PLATFORM
      })
  }

  /**
   * Run handler.
   *
   * @param  {(Function|Object)} handler - Action handler to be ran
   * @returns {Function
   */
  async run (handler) {
    this.setHandler(handler)

    await this._onInit()

    return async (message, context) => {
      this.addContext({ awsContext: context })

      await this.beforeHandle()

      try {
        const event = await this.inputMapper.map({ message, context })

        const result = isFunction(handler)
          ? await handler(event, this.context)
          : await handler[this._getHandlerMethod()](event, this.context)

        const output = await this.outputMapper.map({ message, context, event, result })

        return await output.send()
      } catch (error) {
        return this.#handleError(error, message)
      } finally {
        await this._onTerminate()
      }
    }
  }

  /**
   * Method that runs before running the action handler.
   *
   * @throws {RuntimeError}
   */
  async _onInit () {
    if (isBrowser()) {
      throw new RuntimeError('This `AWSLambdaHTTPAdapter` must be used only in node.js context.')
    }

    await super._onInit()
  }

  #handleError (error, event) {
    this.isDebug && console.error(error)

    const contentType = mime.getType(accepts(event).type(['json', 'html']) ?? 'txt')

    return {
      statusCode: 500,
      statusMessage: statuses.message[500],
      headers: { 'Content-Type': contentType }
    }
  }
}
