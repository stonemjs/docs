import { Config } from '@stone-js/config'
import { Container } from '@stone-js/service-container'
import { IncomingEvent, isClass, isFunction, isObject } from '@stone-js/common'

/**
 * Class representing an Adapter.
 *
 * @author Mr. Stone <evensstone@gmail.com>
 */
export class Adapter {
  #url
  #hooks
  #config
  #context
  #handler
  #container
  #inputMapper
  #outputMapper
  #errorHandler

  /**
   * Create an Apapter.
   *
   * @param  {adapterOptions} [options={}] - Apapter configuration options
   * @returns {Adapter}
   */
  static create (options = {}) {
    return new this(options)
  }

  /**
   * Create an Apapter and run it.
   *
   * @param  {(Function|Object)} handler - Action handler to be ran by adapter
   * @param  {adapterOptions} [options={}] - Apapter configuration options
   * @returns {*}
   */
  static createAndRun (handler, options = {}) {
    return this.create(options).run(handler)
  }

  /**
   * Create an Apapter.
   *
   * @param {adapterOptions} [options={}] - Apapter configuration options
   */
  constructor (options = {}) {
    this.#hooks = {}
    this.#url = new URL('', options.baseUrl ?? 'http://localhost:8080')

    this.#registerBindings(options)
  }

  /** @returns {Config} */
  get config () {
    return this.#config
  }

  /** @returns {Container} */
  get container () {
    return this.#container
  }

  /** @returns {Object} */
  get context () {
    return this.#context
  }

  /** @returns {(Function|Object)} */
  get handler () {
    return this.#handler
  }

  /** @returns {Object} */
  get errorHandler () {
    return this.#errorHandler
  }

  /**
   * Mapper for input data transformation.
   *
   * @returns {Mapper}
   * @throws {TypeError}
   */
  get inputMapper () {
    if (!this.#inputMapper) {
      throw new TypeError('No input mapper instance has been provided.')
    }

    return this.#inputMapper
  }

  /**
   * Mapper for output data transformation.
   *
   * @returns {Mapper}
   * @throws {TypeError}
   */
  get outputMapper () {
    if (!this.#outputMapper) {
      throw new TypeError('No output mapper instance has been provided.')
    }

    return this.#outputMapper
  }

  /** @returns {boolean} */
  get isDebug () {
    return this.#config.get('debug', false)
  }

  /** @returns {string} */
  get port () {
    return this.#url.port
  }

  /** @returns {string} */
  get host () {
    return this.#url.href
  }

  /** @returns {string} */
  get hostname () {
    return this.#url.hostname
  }

  /** @returns {string} */
  get protocol () {
    return this.#url.protocol.replace(':', '')
  }

  /** @returns {boolean} */
  get isSecure () {
    return this.protocol === 'https'
  }

  /**
   * Lifecycle hooks listener.
   *
   * @callback hookListener
   * @param {Object} context - Current Adapter execution context.
   * @param {*} [response] - Only for onTerminate hook.
   */

  /**
   * Register event listener for lifecycle hooks.
   *
   * @param  {('onInit'|'beforeHandle'|'onTerminate')} event
   * @param  {hookListener} listener
   * @returns {this}
   */
  hook (event, listener) {
    this.#hooks[event] ??= []
    this.#hooks[event].push(listener)
    return this
  }

  /**
   * Run handler.
   *
   * @param  {(Function|Object)} handler - Action handler to be ran.
   * @returns {*}
   * @throws {TypeError}
   */
  async run (handler) {
    this.setHandler(handler)

    await this._onInit()

    const event = await this._getEvent()

    await this.beforeHandle()

    const output = isFunction(handler)
      ? await handler(event, this.#context)
      : await handler[this._getHandlerMethod()](event, this.#context)

    await this._onTerminate()

    return output
  }

  /**
   * Register service to container.
   *
   * @param  {Function} Service
   * @param  {boolean}  singleton
   * @param  {string[]} [alias=[]]
   * @returns {this}
   */
  registerService (Service, singleton, alias = []) {
    if (isClass(Service)) {
      this.#container.autoBinding(Service, Service, singleton, alias)
    }
    return this
  }

  /**
   * Register services to container.
   *
   * @param  {Function[]} Services
   * @param  {boolean}  [singleton=true]
   * @returns {this}
   */
  registerServices (Services, singleton = true) {
    Services.forEach(Service => this.registerService(Service, singleton))
    return this
  }

  /**
   * Set hanlder.
   *
   * @param  {(Function|Object)} handler
   * @returns {this}
   */
  setHandler (handler) {
    this.#handler = handler
    return this
  }

  /**
   * Add context.
   *
   * @param  {Object} context
   * @returns {this}
   */
  addContext (context) {
    this.#context = { ...this.#context, ...context }
    return this
  }

  /**
   * Set input mapper.
   *
   * @param  {Mapper} mapper
   * @returns {this}
   */
  setInputMapper (mapper) {
    this.registerServices(this.#config.get('middleware.input', []))
    this.#inputMapper = mapper.addMiddleware(this.#config.get('middleware.input', []))
    return this
  }

  /**
   * Set output mapper.
   *
   * @param  {Mapper} mapper
   * @returns {this}
   */
  setOutputMapper (mapper) {
    this.registerServices(this.#config.get('middleware.output', []))
    this.#outputMapper = mapper.addMiddleware(this.#config.get('middleware.output', []))
    return this
  }

  /**
   * Create and return the event.
   *
   * @returns {IncomingEvent}
   */
  _getEvent () {
    return IncomingEvent.create({ url: this.#url })
  }

  /**
   * Handler method name.
   *
   * @returns {string}
   */
  _getHandlerMethod () {
    return this.#config.get('handler', 'handle')
  }

  /**
   * Method that runs before everything.
   *
   * @throws {TypeError}
   */
  async _onInit () {
    if (!isFunction(this.#handler) && !isObject(this.#handler)) {
      throw new TypeError('The `handler` must be a function or an object.')
    }

    if (Array.isArray(this.#hooks.onInit)) {
      for (const listener of this.#hooks.onInit) {
        await listener(this.#context)
      }
    }

    if (isFunction(this.#handler.onInit)) {
      await this.#handler.onInit(this.#context)
    }
  }

  /**
   * Method that runs before running the action handler.
   */
  async beforeHandle () {
    if (Array.isArray(this.#hooks.beforeHandle)) {
      for (const listener of this.#hooks.beforeHandle) {
        await listener(this.#context)
      }
    }

    if (isFunction(this.#handler.beforeHandle)) {
      await this.#handler.beforeHandle(this.#context)
    }
  }

  /**
   * Method that runs after running the action handler.
   */
  async _onTerminate () {
    if (Array.isArray(this.#hooks.onTerminate)) {
      for (const listener of this.#hooks.onTerminate) {
        await listener(this.#context)
      }
    }

    if (isFunction(this.#handler.onTerminate)) {
      await this.#handler.onTerminate(this.#context)
    }
  }

  #registerBindings (options) {
    this.#container = new Container()
    this.#config = new Config(options)
    this.#context = {
      config: this.#config
    }

    this
      .#container
      .instance(Container, this.#container)
      .instance(Config, this.#config)
      .alias(Container, 'container')
      .alias(Config, 'config')

    return this
  }
}
